EAT = ZoneInfo("Africa/Dar_es_Salaam")


def eat_now() -> datetime:
    """Return timezone-aware East Africa Time on every device/session."""
    return datetime.now(EAT)


def _firebase_service_account() -> dict:
    """Load Firebase service-account JSON without committing the key to GitHub."""
    if "firebase_service_account" in st.secrets:
        return dict(st.secrets["firebase_service_account"])

    json_text = os.getenv("FIREBASE_SERVICE_ACCOUNT_JSON", "").strip()
    if json_text:
        return json.loads(json_text)

    json_path = os.getenv("GOOGLE_APPLICATION_CREDENTIALS", "").strip()
    if json_path and os.path.exists(json_path):
        with open(json_path, "r", encoding="utf-8") as file:
            return json.load(file)

    raise RuntimeError(
        "Firebase service account is missing. Add [firebase_service_account] "
        "to .streamlit/secrets.toml or set FIREBASE_SERVICE_ACCOUNT_JSON."
    )


@st.cache_resource(show_spinner=False)
def firebase_services():
    service_account = _firebase_service_account()
    if not firebase_admin._apps:
        firebase_admin.initialize_app(credentials.Certificate(service_account))
    return firestore.client(), auth


def firebase_web_api_key() -> str:
    if "FIREBASE_WEB_API_KEY" in st.secrets:
        return str(st.secrets["FIREBASE_WEB_API_KEY"])
    key = os.getenv("FIREBASE_WEB_API_KEY", "").strip()
    if not key:
        raise RuntimeError("FIREBASE_WEB_API_KEY is missing.")
    return key


def _auth_rest(endpoint: str, payload: dict) -> dict:
    url = f"https://identitytoolkit.googleapis.com/v1/accounts:{endpoint}?key={firebase_web_api_key()}"
    response = requests.post(url, json=payload, timeout=20)
    data = response.json()
    if not response.ok:
        code = data.get("error", {}).get("message", "FIREBASE_AUTH_ERROR")
        friendly = {
            "EMAIL_EXISTS": "This email is already registered.",
            "EMAIL_NOT_FOUND": "Invalid email or password.",
            "INVALID_PASSWORD": "Invalid email or password.",
            "INVALID_LOGIN_CREDENTIALS": "Invalid email or password.",
            "WEAK_PASSWORD : Password should be at least 6 characters": "Password must contain at least 6 characters.",
            "USER_DISABLED": "This account has been disabled.",
        }
        raise ValueError(friendly.get(code, code.replace("_", " ").title()))
    return data


def create_user(full_name: str, email: str, password: str, role: str = "user"):
    full_name = full_name.strip()
    email = email.strip().lower()
    if len(full_name) < 2:
        raise ValueError("Enter a valid full name.")
    if "@" not in email or "." not in email.split("@")[-1]:
        raise ValueError("Enter a valid email address.")
    if len(password) < 6:
        raise ValueError("Password must contain at least 6 characters.")

    result = _auth_rest("signUp", {
        "email": email,
        "password": password,
        "returnSecureToken": True,
    })
    uid = result["localId"]
    db, _ = firebase_services()
    db.collection("users").document(uid).set({
        "uid": uid,
        "fullName": full_name,
        "username": full_name,
        "email": email,
        "role": role,
        "createdAt": firestore.SERVER_TIMESTAMP,
        "createdAtEAT": eat_now().isoformat(),
        "lastLoginAt": None,
    }, merge=True)
    return uid


def authenticate_user(email: str, password: str):
    result = _auth_rest("signInWithPassword", {
        "email": email.strip().lower(),
        "password": password,
        "returnSecureToken": True,
    })
    token = result["idToken"]
    db, firebase_auth = firebase_services()
    decoded = firebase_auth.verify_id_token(token)
    uid = decoded["uid"]
    ref = db.collection("users").document(uid)
    snapshot = ref.get()
    profile = snapshot.to_dict() if snapshot.exists else {}
    full_name = profile.get("fullName") or profile.get("username") or result.get("displayName") or email.split("@")[0]
    role = profile.get("role", "user")
    ref.set({
        "uid": uid,
        "fullName": full_name,
        "username": full_name,
        "email": result.get("email", email).lower(),
        "role": role,
        "lastLoginAt": firestore.SERVER_TIMESTAMP,
        "lastLoginAtEAT": eat_now().isoformat(),
    }, merge=True)
    return {
        "id": uid,
        "full_name": full_name,
        "email": result.get("email", email).lower(),
        "role": role,
        "id_token": token,
    }


def save_prediction_record(record: dict):
    db, _ = firebase_services()
    signal_text = str(record["final_signal"])
    signal = "BUY" if "BUY" in signal_text else "SELL" if "SELL" in signal_text else "HOLD"
    payload = {
        "userId": st.session_state.user_id,
        "uid": st.session_state.user_id,
        "userName": st.session_state.user_name,
        "username": st.session_state.user_name,
        "userEmail": st.session_state.user_email,
        "email": st.session_state.user_email,
        "stock": record["stock"],
        "forecastPeriod": record["period"],
        "period": record["period"],
        "currentPrice": float(record["current_price"]),
        "forecastPrice": float(record["final_price"]),
        "changePercent": float(record["final_change"]),
        "recommendation": signal,
        "signal": signal,
        "signalDisplay": signal_text,
        "confidence": int(record["final_confidence"]),
        "predictionTime": firestore.SERVER_TIMESTAMP,
        "predictionTimeEAT": record["now"].astimezone(EAT).isoformat(),
        "forecastTime": record["forecast_time"].astimezone(EAT).isoformat(),
        "timezone": "Africa/Dar_es_Salaam",
    }
    doc = db.collection("predictions").document()
    payload["predictionId"] = doc.id
    doc.set(payload)


def _format_firestore_time(value):
    if value is None:
        return ""
    if hasattr(value, "astimezone"):
        return value.astimezone(EAT).strftime("%d %b %Y %H:%M:%S EAT")
    return str(value)


def load_admin_data():
    db, _ = firebase_services()
    users_rows = []
    for doc in db.collection("users").stream():
        item = doc.to_dict() or {}
        users_rows.append({
            "UID": doc.id,
            "Name": item.get("fullName") or item.get("username", ""),
            "Email": item.get("email", ""),
            "Role": item.get("role", "user"),
            "Registered": _format_firestore_time(item.get("createdAt")) or item.get("createdAtEAT", ""),
            "Last Login": _format_firestore_time(item.get("lastLoginAt")) or item.get("lastLoginAtEAT", ""),
        })

    prediction_rows = []
    for doc in db.collection("predictions").stream():
        item = doc.to_dict() or {}
        recommendation = item.get("recommendation") or item.get("signal", "HOLD")
        prediction_rows.append({
            "Prediction ID": doc.id,
            "User": item.get("userName") or item.get("username", ""),
            "Email": item.get("userEmail") or item.get("email", ""),
            "Stock": item.get("stock", ""),
            "Period": item.get("forecastPeriod") or item.get("period", ""),
            "Current Price": item.get("currentPrice", 0),
            "Forecast Price": item.get("forecastPrice", 0),
            "Change %": item.get("changePercent", 0),
            "Recommendation": recommendation,
            "Confidence": item.get("confidence", 0),
            "Prediction Time": _format_firestore_time(item.get("predictionTime")) or item.get("predictionTimeEAT", ""),
            "Forecast Time": item.get("forecastTime", ""),
            "_sort_time": item.get("predictionTime") or item.get("predictionTimeEAT", ""),
        })

    users = pd.DataFrame(users_rows)
    predictions = pd.DataFrame(prediction_rows)
    if not predictions.empty:
        predictions = predictions.sort_values("_sort_time", ascending=False).drop(columns=["_sort_time"])
    return users, predictions

def create_features(dataframe: pd.DataFrame) -> pd.DataFrame:
    frame = dataframe.copy()
    frame["Return"] = frame["Close"].pct_change()
    frame["MA10"] = frame["Close"].rolling(10).mean()
    frame["MA20"] = frame["Close"].rolling(20).mean()
    frame["MA50"] = frame["Close"].rolling(50).mean()
    frame["Volatility"] = frame["Return"].rolling(10).std()
    frame["Momentum"] = frame["Close"] - frame["Close"].shift(10)

    delta = frame["Close"].diff()
    gain = delta.clip(lower=0)
    loss = -delta.clip(upper=0)
    avg_gain = gain.rolling(14).mean()
    avg_loss = loss.rolling(14).mean().replace(0, np.nan)
    rs = avg_gain / avg_loss
    frame["RSI"] = 100 - (100 / (1 + rs))

    ema12 = frame["Close"].ewm(span=12, adjust=False).mean()
    ema26 = frame["Close"].ewm(span=26, adjust=False).mean()
    frame["MACD"] = ema12 - ema26
    frame["Signal"] = frame["MACD"].ewm(span=9, adjust=False).mean()
    return frame.dropna().reset_index(drop=True)


@st.cache_resource(show_spinner=False)
def load_assets():
    scaler_obj = joblib.load("models/scaler.pkl")

def forecast_days(selected_period: str) -> int:
    return {"Today": 1, "Tomorrow": 1, "Next Week": 7, "Next Month": 30}[selected_period]


def next_business_day(day):
    while day.weekday() >= 5:
        day += timedelta(days=1)
    return day


def create_forecast_times(selected_period: str, count: int, now: datetime, rng=None) -> list[datetime]:
    """Create deterministic EAT forecast times, identical across device time zones."""
    if selected_period == "Today":
        return [now + timedelta(hours=3)]

    if selected_period == "Tomorrow":
        day = next_business_day(now.date() + timedelta(days=1))
        return [datetime.combine(day, datetime.min.time(), tzinfo=EAT).replace(hour=11)]

    results = []
    day = now.date() + timedelta(days=1)
    while len(results) < count:
        day = next_business_day(day)
        results.append(datetime.combine(day, datetime.min.time(), tzinfo=EAT).replace(hour=11))
        day += timedelta(days=1)
    return results


def inverse_close_values(values, scaler_obj, close_index: int, feature_count: int) -> list[float]:
    result = []
    for value in values:
        row = np.zeros((1, feature_count))
        row[0, close_index] = value
        result.append(float(scaler_obj.inverse_transform(row)[0][close_index]))
    return result


def predict_scaled_path(days, scaled_data, feature_names, prepared_df, onnx_session, keras_obj, rng):
    if len(scaled_data) < 60:
        raise ValueError("At least 60 prepared rows are required.")

    sequence = scaled_data[-60:].copy()
    close_index = feature_names.index("Close")
    results = []

    recent_return = float(prepared_df["Return"].tail(5).mean())
    trend = float((prepared_df["MA10"].iloc[-1] - prepared_df["MA20"].iloc[-1]) / prepared_df["Close"].iloc[-1])

    for step in range(days):
        x = sequence.reshape(1, 60, scaled_data.shape[1]).astype(np.float32)

        if onnx_session is not None:
            input_name = onnx_session.get_inputs()[0].name
            raw = float(np.asarray(onnx_session.run(None, {input_name: x})[0]).reshape(-1)[0])
        elif keras_obj is not None:
            raw = float(np.asarray(keras_obj.predict(x, verbose=0)).reshape(-1)[0])
        else:
            raise RuntimeError(t("model_missing"))

        ai_component = float(np.tanh(raw)) * 0.004
        technical_component = float(np.clip(recent_return * 0.25 + trend * 0.45, -0.004, 0.004))
        wave = np.sin((step + 1) * 1.2) * 0.0014
        noise = float(rng.normal(0, 0.0015 if days > 1 else 0.0010))
        change = float(np.clip(ai_component + technical_component + wave + noise, -0.012, 0.012))

        next_close = float(sequence[-1, close_index] * (1 + change))
        results.append(next_close)
        new_row = sequence[-1].copy()
        new_row[close_index] = next_close
        sequence = np.vstack([sequence[1:], new_row])

    return np.asarray(results)


def recommendation(price, reference, previous, prepared_df, index, rng):
    change_pct = (price - reference) / reference * 100
    step_pct = 0 if previous is None else (price - previous) / previous * 100
    rsi = float(prepared_df["RSI"].iloc[-1])
    macd = float(prepared_df["MACD"].iloc[-1])
    macd_signal = float(prepared_df["Signal"].iloc[-1])
    ma10 = float(prepared_df["MA10"].iloc[-1])
    ma20 = float(prepared_df["MA20"].iloc[-1])
    volatility = float(prepared_df["Volatility"].iloc[-1])

    bullish = 0.0
    bearish = 0.0
    reasons = []
    threshold = 0.18 + min(index, 10) * 0.015

    if change_pct > threshold:
        bullish += 2.8
        reasons.append(f"price rises {change_pct:.2f}%")
    elif change_pct < -threshold:
        bearish += 2.8
        reasons.append(f"price falls {abs(change_pct):.2f}%")
    else:
        reasons.append(f"movement {change_pct:+.2f}% is near neutral")

    if ma10 > ma20:
        bullish += 0.9
        reasons.append("moving-average trend is positive")
    elif ma10 < ma20:
        bearish += 0.9
        reasons.append("moving-average trend is negative")

    if macd > macd_signal:
        bullish += 0.8
        reasons.append("MACD is bullish")
    elif macd < macd_signal:
        bearish += 0.8
        reasons.append("MACD is bearish")

    if rsi < 35:
        bullish += 0.7
        reasons.append("RSI suggests oversold conditions")
    elif rsi > 70:
        bearish += 0.7
        reasons.append("RSI suggests overbought conditions")

    if step_pct > threshold * 0.45:
        bullish += 0.8
    elif step_pct < -threshold * 0.45:
        bearish += 0.8

    pulse = float(rng.normal(0, 0.55))
    if pulse > 0.35:
        bullish += 0.55
    elif pulse < -0.35:
        bearish += 0.55

    gap = bullish - bearish
    if bullish >= 2.6 and gap >= 0.65:
        signal = "BUY 🟢"
    elif bearish >= 2.6 and gap <= -0.65:
        signal = "SELL 🔴"
    else:
        signal = "HOLD 🟡"

    confidence = int(np.clip(56 + abs(gap) * 7 + min(abs(change_pct), 2.5) * 5, 52, 95))
    if volatility > 0.03:
        confidence = max(50, confidence - 5)

    return signal, change_pct, confidence, "; ".join(reasons)

def build_csv(summary_df: pd.DataFrame, schedule_df: pd.DataFrame) -> bytes:
    text = "PREDICTION SUMMARY\n"
    text += summary_df.to_csv(index=False)
    text += "\nFORECAST SCHEDULE\n"
    text += schedule_df.to_csv(index=False)
    return text.encode("utf-8-sig")


def build_excel(summary_df: pd.DataFrame, schedule_df: pd.DataFrame) -> bytes:
    buffer = BytesIO()
    with pd.ExcelWriter(buffer, engine="openpyxl") as writer:
        summary_df.to_excel(writer, sheet_name="Summary", index=False)
        schedule_df.to_excel(writer, sheet_name="Forecast", index=False)
        for sheet_name in ["Summary", "Forecast"]:
            sheet = writer.sheets[sheet_name]
            for column_cells in sheet.columns:
                width = max(len(str(cell.value or "")) for cell in column_cells) + 2
                sheet.column_dimensions[column_cells[0].column_letter].width = min(width, 45)
    buffer.seek(0)
    return buffer.getvalue()

def build_zip(files: dict[str, bytes]) -> bytes:
    buffer = BytesIO()
    with zipfile.ZipFile(buffer, "w", zipfile.ZIP_DEFLATED) as archive:
        for filename, content in files.items():
            archive.writestr(filename, content)
    buffer.seek(0)
    return buffer.getvalue()
